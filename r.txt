  <script>
    const placeholder = document.getElementById("report-placeholder");
    const params = new URLSearchParams(window.location.search);
    const rid = params.get("rid");

    if (!rid) {
      placeholder.innerHTML = "<p style='text-align:center;'>Invalid report link!</p>";
    } else {
      (async () => {
        try {
          const resp = await fetch(`/api/getReport?rid=${encodeURIComponent(rid)}`);
          const data = await resp.json();

          if (!resp.ok || !data.html) {
            throw new Error(data.error || "Report not found");
          }

          // Replace placeholder with report HTML
          placeholder.innerHTML = data.html;
        } catch (err) {
          console.error("Error loading report:", err);
          placeholder.innerHTML = "<p style='text-align:center;'>Error loading report.</p>";
        }
      })();
    }
  </script>
